问题一

解决：自连接&多表删除
delete s1 from scores s1
join scores s2
on s1.student_id=s2.student_id
And s1………..=s2………..
And s1.id < s2.id

问题2.1找一个老师的所有学生
select
count()
from表一
join表二on 表一.主键=表二.外键
join表三on 表一.主键=表三.外键
where表一.主键=”老师姓名”
group by 表一.主键

问题2.2算出一个老师所有学生的平均分
select
avg()
from表一
join表二on 表一.主键=表二.外键
join表三on 表一.主键=表三.外键
where表一.主键=”老师姓名”
group by 表一.主键

问题2.3找一个老师所有学生的最高分
select
from表一
join表二on 表一.主键=表二.外键
join表三on 表一.主键=表三.外键
where表一.主键=”老师姓名”
group by 表一.主键
order by 表三.成绩 desc
limit 1
